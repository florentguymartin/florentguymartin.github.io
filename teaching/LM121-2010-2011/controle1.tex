\title{Controle 1 LM 121}
\maketitle
\begin{enumerate}

\item
Décrire géométriquement les $z\in \mathbb{C}$ (avec $z\neq - i$) tels que 
$\frac{z+1}{z+i} \in \mathbb{R}$.

\item
Trouver les solutions dans $\mathbb{C}$ de l'équation
$$z^2 +(3i-5)z+16-11i =0$$

\item
Soit $f$ la rotation de centre $(0,0)$ et d'angle $\frac{\pi}{2}$, et $g$ la rotation de centre $(1,2)$ et d'angle $\frac{\pi}{2}$. Identifier géométriquement $g \circ f$.

\item
Soit $z\in \mathbb{C}$ tel que $1+z^2+z^4+z^6+z^8=0$. Que vaut $|z|$ ?

\item 
Soit $u=(1,2,3) , \ v=(-1,1,2) ,\ w=(3,3,4)$ et $z=(1,1,1)$ des vecteurs de $\mathbb{R}^3$. $u, v$ et $w$ sont-ils liés? $z$ est-il combinaison linéaire de $u,v$ et $w$?

\end{enumerate}
