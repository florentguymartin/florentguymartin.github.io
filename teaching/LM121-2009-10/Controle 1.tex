\begin{enumerate}
 \item[1]
Soit $a\in \mathbb{C}^*$. Montrer qu'il existe exactement deux nombres complexes $z$ tels que $z^2 =a$.


\item[2]
Donner les racines carrées de $2i -5$.

\item[3]
soit $u=(-2,1,0)$ , \ $v=(3,0,1)$ et $w=(1,1,t)$ où $t\in \mathbb{R}$. Pour quel(s) $t$ ces trois vecteurs sont liés?

\item[4]
\begin{enumerate}
 \item Soit $\mathcal{D}$ la droite d'équation parametrique
\begin{align*}
 x&=2+t \\
y & =3+2t \\
z & =1-t 
\end{align*}
Donner une équation cartésienne de $\mathcal{D}$.
\item
Soit $\mathcal{D'}$ la droite d'équation parametrique
\begin{align*}
 x &=2t \\
y & =1-t \\
z & = 2+2t
\end{align*}
$\mathcal{D}$ et $\mathcal{D'}$ ont-elles des points en commun? (dit autrement, a-t-on $\mathcal{D} \cap \mathcal{D'} =\emptyset$?)

\end{enumerate}


\end{enumerate}
